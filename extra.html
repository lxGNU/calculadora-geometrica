💡 Método para sólidos irregulares

Principio de Arquímedes:

Volumen irregular = Volumen final - Volumen inicial
Procedimiento:
1. Llenar recipiente con agua (V₁)
2. Sumergir objeto completamente
3. Medir nuevo volumen (V₂)
4. V_objeto = V₂ - V₁

🐚 Script en Bash

#!/bin/bash

# Evidencia GA2-240201528-AA4-EV01
# Algoritmo para cálculo de áreas y volúmenes

mostrar_introduccion() {
    clear
    echo "=============================================="
    echo "  ALGORITMO PARA CÁLCULO DE ÁREAS Y VOLÚMENES"
    echo "  Evidencia GA2-240201528-AA4-EV01"
    echo "=============================================="
    echo ""
    echo "Figuras planas disponibles:"
    echo "• Cuadrado: Área = L², Perímetro = 4L"
    echo "• Círculo: Área = πr², Perímetro = 2πr"
    echo ""
    echo "Sólidos regulares disponibles:"
    echo "• Cubo: Volumen = L³"
    echo "• Esfera: Volumen = 4/3πr³"
    echo ""
}

calcular_cuadrado() {
    read -p "Ingrese el lado: " lado
    if [[ $lado =~ ^[0-9]+([.][0-9]+)?$ ]] && (( $(echo "$lado > 0" | bc -l) )); then
        area=$(echo "$lado * $lado" | bc -l)
        perimetro=$(echo "4 * $lado" | bc -l)
        printf "Área: %.2f | Perímetro: %.2f\n" $area $perimetro
    else
        echo "Error: Valor positivo requerido"
    fi
}

calcular_circulo() {
    read -p "Ingrese el radio: " radio
    pi=3.14159265359
    if [[ $radio =~ ^[0-9]+([.][0-9]+)?$ ]] && (( $(echo "$radio > 0" | bc -l) )); then
        area=$(echo "$pi * $radio * $radio" | bc -l)
        perimetro=$(echo "2 * $pi * $radio" | bc -l)
        printf "Área: %.2f | Perímetro: %.2f\n" $area $perimetro
    else
        echo "Error: Valor positivo requerido"
    fi
}

calcular_cubo() {
    read -p "Ingrese el lado: " lado
    if [[ $lado =~ ^[0-9]+([.][0-9]+)?$ ]] && (( $(echo "$lado > 0" | bc -l) )); then
        volumen=$(echo "$lado * $lado * $lado" | bc -l)
        printf "Volumen: %.2f\n" $volumen
    else
        echo "Error: Valor positivo requerido"
    fi
}

calcular_esfera() {
    read -p "Ingrese el radio: " radio
    pi=3.14159265359
    if [[ $radio =~ ^[0-9]+([.][0-9]+)?$ ]] && (( $(echo "$radio > 0" | bc -l) )); then
        volumen=$(echo "scale=4; (4/3) * $pi * ($radio * $radio * $radio)" | bc -l)
        printf "Volumen: %.2f\n" $volumen
    else
        echo "Error: Valor positivo requerido"
    fi
}

metodo_irregular() {
    echo ""
    echo "📌 MÉTODO PARA SÓLIDOS IRREGULARES"
    echo "PRINCIPIO DE ARQUÍMEDES:"
    echo "1. Registrar volumen inicial de agua (V₁)"
    echo "2. Sumergir el objeto completamente"
    echo "3. Registrar volumen final (V₂)"
    echo "4. Volumen del objeto = V₂ - V₁"
    echo ""
}

conclusiones() {
    clear
    echo "════════════════════════════════"
    echo "CONCLUSIONES"
    echo "════════════════════════════════"
    echo "✓ Se implementaron las 4 figuras requeridas"
    echo "✓ Validación de entradas positivas"
    echo "✓ Método de Arquímedes para irregulares"
    echo "✓ Interfaz interactiva y amigable"
    echo "════════════════════════════════"
}

main() {
    mostrar_introduccion
    while true; do
        echo ""
        echo "1. Cuadrado"
        echo "2. Círculo"
        echo "3. Cubo"
        echo "4. Esfera"
        echo "5. Método irregulares"
        echo "6. Conclusiones"
        echo "7. Salir"
        read -p "Seleccione: " opcion
        
        case $opcion in
            1) calcular_cuadrado ;;
            2) calcular_circulo ;;
            3) calcular_cubo ;;
            4) calcular_esfera ;;
            5) metodo_irregular ;;
            6) conclusiones ;;
            7) echo "¡Hasta luego!"; exit 0 ;;
            *) echo "Opción no válida" ;;
        esac
        echo ""
        read -p "Presione Enter para continuar..."
    done
}

main
